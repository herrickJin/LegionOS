# 技术方案：2. 数据库设计（MVP版）

> **对应PRD：** 第4章 领域模型 + 第6章 6.6 数据架构
>
> **目标：** 支撑4周MVP，5个服务、4个数据库实例
>
> **核心原则：** Schema隔离、本地事务优先、所有表含org_id

---

## 1. 数据库实例规划

| 实例 | 服务 | 说明 |
|------|------|------|
| **core_db** | core-svc | 经营决策+项目管理+项目实施+产研流程，Schema隔离 |
| **iam_db** | iam-svc | 用户权限，独立实例 |
| **asset_db** | asset-svc | 技术资产+知识经验+自我进化+价值评估 |
| **agent_db** | agent-svc | 数字员工 |
| **MongoDB** | core-svc + asset-svc | MRD文档、知识文档、改进需求 |
| **Redis** | 全局 | 缓存、运行时状态、项目上下文 |

---

## 2. core_db 设计

### 2.1 Schema划分

```sql
-- 创建Schema
CREATE SCHEMA IF NOT EXISTS opp_schema;
CREATE SCHEMA IF NOT EXISTS prj_schema;
CREATE SCHEMA IF NOT EXISTS tsk_schema;
CREATE SCHEMA IF NOT EXISTS wf_schema;

-- 创建用户（各服务独立）
CREATE USER 'opp_user'@'%' IDENTIFIED BY 'opp_pass';
CREATE USER 'prj_user'@'%' IDENTIFIED BY 'prj_pass';
CREATE USER 'tsk_user'@'%' IDENTIFIED BY 'tsk_pass';
CREATE USER 'wf_user'@'%' IDENTIFIED BY 'wf_pass';

-- 授权（只限本Schema）
GRANT ALL PRIVILEGES ON opp_schema.* TO 'opp_user'@'%';
GRANT ALL PRIVILEGES ON prj_schema.* TO 'prj_user'@'%';
GRANT ALL PRIVILEGES ON tsk_schema.* TO 'tsk_user'@'%';
GRANT ALL PRIVILEGES ON wf_schema.* TO 'wf_user'@'%';
```

### 2.2 opp_schema（经营决策域）

```sql
-- 商机表
CREATE TABLE opp_opportunity (
    id VARCHAR(64) PRIMARY KEY COMMENT '商机ID',
    name VARCHAR(256) NOT NULL COMMENT '商机名称',
    status VARCHAR(32) NOT NULL DEFAULT '发现' COMMENT '状态：发现/评估中/可行/不可行/已归档',
    source VARCHAR(64) COMMENT '来源：内部需求/客户反馈/市场调研',
    priority INT DEFAULT 3 COMMENT '优先级 1-5',
    estimated_value DECIMAL(18,2) COMMENT '预估价值',
    org_id VARCHAR(64) NOT NULL COMMENT '组织ID',
    created_by VARCHAR(64) NOT NULL COMMENT '创建人',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_status(status),
    INDEX idx_org_id(org_id),
    INDEX idx_created_at(created_at)
) COMMENT='商机';

-- 可行性报告表
CREATE TABLE opp_feasibility_report (
    id VARCHAR(64) PRIMARY KEY,
    opportunity_id VARCHAR(64) NOT NULL COMMENT '关联商机',
    status VARCHAR(32) NOT NULL DEFAULT '编写中' COMMENT '编写中/评审中/已通过/已拒绝',
    technical_feasibility TEXT COMMENT '技术可行性分析',
    resource_requirement TEXT COMMENT '资源需求评估',
    risk_analysis TEXT COMMENT '风险分析',
    conclusion VARCHAR(32) COMMENT '结论：可行/不可行/有条件可行',
    org_id VARCHAR(64) NOT NULL,
    created_by VARCHAR(64) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (opportunity_id) REFERENCES opp_opportunity(id),
    INDEX idx_opportunity_id(opportunity_id),
    INDEX idx_status(status)
) COMMENT='可行性报告';

-- MRD表
CREATE TABLE opp_mrd (
    id VARCHAR(64) PRIMARY KEY COMMENT 'MRD ID',
    name VARCHAR(256) NOT NULL COMMENT 'MRD名称',
    opportunity_id VARCHAR(64) COMMENT '关联商机',
    status VARCHAR(32) NOT NULL DEFAULT '草稿' COMMENT '草稿/评审中/已通过/已拒绝/已归档',
    content TEXT COMMENT 'MRD内容（JSON或Markdown）',
    priority INT DEFAULT 3 COMMENT '优先级',
    org_id VARCHAR(64) NOT NULL,
    created_by VARCHAR(64) NOT NULL,
    approved_by VARCHAR(64) COMMENT '审批人',
    approved_at DATETIME COMMENT '审批时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (opportunity_id) REFERENCES opp_opportunity(id),
    INDEX idx_status(status),
    INDEX idx_org_id(org_id),
    INDEX idx_opportunity_id(opportunity_id)
) COMMENT='市场需求文档';
```

## 3. iam_db 设计

```sql
-- 用户表（人 + 数字员工）
CREATE TABLE sys_user (
    id VARCHAR(64) PRIMARY KEY COMMENT '用户ID',
    username VARCHAR(64) NOT NULL UNIQUE COMMENT '用户名',
    password_hash VARCHAR(256) COMMENT '密码哈希（数字员工可为空）',
    user_type VARCHAR(16) NOT NULL DEFAULT 'HUMAN' COMMENT 'HUMAN/AGENT',
    real_name VARCHAR(64) COMMENT '真实姓名/数字员工名称',
    org_id VARCHAR(64) NOT NULL COMMENT '组织ID',
    status VARCHAR(16) DEFAULT '启用' COMMENT '启用/禁用/删除',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_org_id(org_id),
    INDEX idx_user_type(user_type),
    INDEX idx_status(status)
) COMMENT='用户';

-- 角色表
CREATE TABLE sys_role (
    id VARCHAR(64) PRIMARY KEY,
    role_code VARCHAR(64) NOT NULL UNIQUE COMMENT '角色编码',
    role_name VARCHAR(64) NOT NULL COMMENT '角色名称',
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) COMMENT='角色';

-- 用户角色关联
CREATE TABLE sys_user_role (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(64) NOT NULL,
    role_code VARCHAR(64) NOT NULL,
    UNIQUE KEY uk_user_role (user_id, role_code),
    FOREIGN KEY (user_id) REFERENCES sys_user(id),
    FOREIGN KEY (role_code) REFERENCES sys_role(role_code)
) COMMENT='用户角色';

-- 权限表
CREATE TABLE sys_permission (
    id VARCHAR(64) PRIMARY KEY,
    perm_code VARCHAR(128) NOT NULL UNIQUE COMMENT '权限编码：domain:action',
    perm_name VARCHAR(128) NOT NULL,
    domain VARCHAR(64) NOT NULL COMMENT '所属域',
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_domain(domain)
) COMMENT='权限';

-- 角色权限关联
CREATE TABLE sys_role_permission (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    role_code VARCHAR(64) NOT NULL,
    perm_code VARCHAR(128) NOT NULL,
    UNIQUE KEY uk_role_perm (role_code, perm_code),
    FOREIGN KEY (role_code) REFERENCES sys_role(role_code),
    FOREIGN KEY (perm_code) REFERENCES sys_permission(perm_code)
) COMMENT='角色权限';

-- 组织表
CREATE TABLE sys_org (
    id VARCHAR(64) PRIMARY KEY,
    name VARCHAR(256) NOT NULL,
    parent_id VARCHAR(64) COMMENT '父组织ID',
    org_path VARCHAR(512) COMMENT '组织路径：org1/org2/org3',
    level INT DEFAULT 1 COMMENT '层级',
    status VARCHAR(16) DEFAULT '启用',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_parent_id(parent_id)
) COMMENT='组织';

-- 数字员工扩展表
CREATE TABLE sys_agent_user (
    id VARCHAR(64) PRIMARY KEY,
    user_id VARCHAR(64) NOT NULL UNIQUE,
    prototype_id VARCHAR(64) COMMENT '关联原型ID',
    instance_id VARCHAR(64) COMMENT '关联实例ID',
    skills JSON COMMENT '技能列表',
    max_concurrent_tasks INT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES sys_user(id)
) COMMENT='数字员工用户扩展';
```

    created_by VARCHAR(64) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_type(type),
    INDEX idx_status(status),
    INDEX idx_org_id(org_id)
) COMMENT='知识条目';

-- 改进需求表
CREATE TABLE evo_improvement (
    id VARCHAR(64) PRIMARY KEY,
    title VARCHAR(512) NOT NULL,
    category VARCHAR(64) COMMENT '数字员工优化/产研流程改进/系统功能增强/体验问题',
    status VARCHAR(32) NOT NULL DEFAULT '待处理' COMMENT '待处理/处理中/已验证/已关闭/已搁置',
    description TEXT,
    solution TEXT,
    priority INT DEFAULT 3,
    org_id VARCHAR(64) NOT NULL,
    created_by VARCHAR(64) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_status(status),
    INDEX idx_category(category)
) COMMENT='改进需求';

-- 价值积分表
CREATE TABLE val_points (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    owner_id VARCHAR(64) NOT NULL COMMENT '积分所有者（用户/资产/知识ID）',
    owner_type VARCHAR(32) NOT NULL COMMENT 'HUMAN/AGENT/ASSET/KNOWLEDGE',
    points DECIMAL(18,4) NOT NULL DEFAULT 0,
    source_type VARCHAR(64) COMMENT '任务执行/资产复用/知识引用/系统改进',
    source_id VARCHAR(64) COMMENT '来源记录ID',
    description TEXT,
    org_id VARCHAR(64) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_owner(owner_id, owner_type),
    INDEX idx_org_id(org_id)
) COMMENT='价值积分';

-- 价值规则表
CREATE TABLE val_rule (
    id VARCHAR(64) PRIMARY KEY,
    name VARCHAR(256) NOT NULL,
    status VARCHAR(32) NOT NULL DEFAULT '草稿' COMMENT '草稿/待审批/已启用/已停用',
    rule_config JSON NOT NULL COMMENT '规则配置：权重/条件/计算方式',
    version INT DEFAULT 1,
    org_id VARCHAR(64) NOT NULL,
    created_by VARCHAR(64) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_status(status)

task:instance:{instance_id}:current_node → STRING(当前节点ID)
task:instance:{instance_id}:status → STRING(状态)

session:{token} → JSON(用户ID+角色+组织)
